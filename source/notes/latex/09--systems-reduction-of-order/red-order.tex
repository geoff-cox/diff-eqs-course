	\begin{center}  \sc{Differential Equations\\
				\vspace{0.25cm}
				
				Systems and Higher Order DEs}
	\end{center}

	\noindent\rule{7in}{0.4pt} \vspace{0.25cm}
		
	\noindent  What if we have a higher-order DE, and we can't use any of the solution techniques
	we have studied so far?  Is there some way we can leverage this numerical technique to get a 
	solution?  Turns out there is!  We just need to use a simple change of variables called
	\textbf{reduction of order.}

	\medskip

	\begin{enumerate}
		\item  Consider the following IVP:
		
		\[ 
			ty' + y^2 + y''= e^t,\quad y(0) = 4,\quad y'(0) = -3.
		\]
		\smallskip
		
		\begin{enumerate}

			\item What is the independent variable?
			\bigskip

			\item What is the dependent variable?
			\bigskip

			\item What is the order of the DE?
			\bigskip

			\item Is it linear or nonlinear?
			\bigskip
			
			\item Consider the following change of variables:
			\[
				u = y,\qquad v = y'.
			\]
			
			\begin{enumerate}
				\item What is the relationship between $u$ and $v$?
				\vfill
				
				\item Rewrite the DE in terms of the new variables $u$ and $v.$
				\vfill
				
				\item Write the initial conditions in terms of the new variables $u$ and $v.$
				\vfill
			\end{enumerate}

		\end{enumerate}

		\newpage

		\begin{enumerate}

			\item Reduce each of the following to a system of first-order differential equations.
			\bigskip

			\begin{enumerate}
				\item \(z''' + z'' = t^3\)
				\vfill
				
				\item \(x'' + y' = 9\)
				\vfill

				\item \(y'' - 2xy = t\sin t\)
				\vfill
			\end{enumerate}
			
		\end{enumerate}

	\end{enumerate}

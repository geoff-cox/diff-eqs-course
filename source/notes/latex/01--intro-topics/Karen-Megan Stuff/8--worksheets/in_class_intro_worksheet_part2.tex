\item Consider the differential equation
		\[ \frac{dy}{dx} - 1 = e^{y - x}. \]
	
	\ben
	\item Determine if $\ds y = -\ln(-x+C) + x$ where $C$ is a constant, is a solution to the given
		differential equation.\\ \vfill

	\item Is this a general solution or a particular solution?  Explain how you know.\\
		\vspace{1in}
			%General solution.  It contains a constant.
	\een
	
\newpage

\item Consider the differential equation
	\[ \frac{dx}{dt} = \frac{x(x-2)}{2}.\]  
	\ben
	\item Determine if $ \ds x = \frac{2}{1 - Ce^t}$ is a solution to the given
		differential equation.\\ \vfill

	\item Find the particular solution that satisfies $\ds x(0) = \frac{1}{2}.$\\ \vspace{2.5in}
			%\[ x = \frac{2}{1 + 3e^t}\]
	\een

\item Consider $x = c_1\cos(t) + c_2 \sin(t),$ which is a family of
	solutions to the DE $x'' + x = 0.$  Find the particular solution that satisfies $x(\pi/2) = 0,$
	$x'(\pi/2) = 1.$\\ \vfill
			%\[ x = -\cos(t) \]
	
\newpage	

\item Consider $z = c_1e^{3t} + c_2e^{-5t},$ which is the general solution of some second-order DE.
	Find the particular solution that satisfies $z(0) = 2,$
	$z'(0) = 22.$\\ \vfill
			%\[ z = 4e^{3t} - 2e^{-5t} \]

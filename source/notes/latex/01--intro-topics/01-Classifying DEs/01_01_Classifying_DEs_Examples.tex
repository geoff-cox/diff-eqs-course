\noindent Write the differential equation below in the form $y'+p(x)y=q(x)$ and identify $p(x)$ and $q(x)$.  Also, give the order and linearity of the equation.
$$
\D - 4xy' + 5y = \cos (x)\,y - 1
$$

\thispagestyle{empty}
\newpage
\thispagestyle{empty}

\newpage
\noindent  Answers:
\begin{multicols}{2}
\begin{enumerate}[1.]
\item  $\D (1 - x)y'' - 4xy' + 5y = \cos x$
	\begin{itemize}
	\item $y$
	\item $x$
	\item 2nd order
	\item linear
	\end{itemize}

\vspace{1.1cm}

\item  $\D x \frac{d^3y}{dx^3} - \left( \frac{dy}{dx} \right)^4 + y = 0$
	\begin{itemize}
	\item $y$
	\item $t$
	\item 3rd order
	\item nonlinear
	\item $\D \left( \frac{dy}{dx} \right)^4$
	\end{itemize}

\vspace{1.1cm}

\item  $\D t^5 y^{(4)} - t^3 y'' + 6y = 0$
	\begin{itemize}
	\item $x$
	\item $t$
	\item 4th order
	\item linear
	\end{itemize}

\vspace{1.1cm}

\item  $\D \frac{d^2u}{dr^2} + \frac{du}{dr} + u = \cos(r+u) $
	\begin{itemize}
	\item $u$
	\item $r$
	\item 2nd order
	\item nonlinear
	\item $\D \cos(r+u)$
	\end{itemize}

\vspace{1.1cm}

\item  $\D \frac{d^2y}{dx^2} = \sqrt{1 + \left(\D \frac{dy}{dx}  \right)^2} $
	\begin{itemize}
	\item $y$
	\item $x$
	\item 2nd order
	\item nonlinear
	\item $\D \sqrt{1 + \left(\D \frac{dy}{dx}  \right)^2}$
	\end{itemize}

\vspace{1.1cm}

\item  $\D \frac{d^2R}{dt^2} = -\frac{k}{R^2}$
	\begin{itemize}
	\item $R$
	\item $t$
	\item 2nd order 
	\item nonlinear
	\item $\D -\frac{k}{R^2}$
	\end{itemize}

\vspace{1.1cm}

\item  $\D (\sin \theta)y''' - (\cos \theta)y' = 2$
	\begin{itemize}
	\item $y$
	\item $\theta$
	\item 3rd order
	\item linear
	\end{itemize}

\vspace{1.1cm}

\item  $\D \ddot{x} - \left( 1 - \frac{\dot{x}^2}{3} \right)\dot{x} + x = 0$
	\begin{itemize}
	\item $x$
	\item $t$
	\item 2nd order
	\item nonlinear
	\item $\D \left( 1 - \frac{\dot{x}^2}{3} \right)\dot{x}$
	\end{itemize}
\end{enumerate}
\end{multicols}

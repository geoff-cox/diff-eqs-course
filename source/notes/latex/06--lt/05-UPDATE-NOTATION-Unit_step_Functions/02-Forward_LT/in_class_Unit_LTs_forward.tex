\item Cadet C. Onfused is taking the Laplace transform of the function $m(t) = e^{-5t} - 17e^{-5t}U(t-2),$ as follows.
		\begin{align*} f(t)	&= e^{-5t}\\
						&\\
					M(s)	&= \L\{ e^{-5t} - 17e^{-5t}U(t-2) \}\\
						&= \L\{ e^{-5t}\} - 17\L\{ e^{-5t}U(t-2) \}\\
						&= \L\{ e^{-5t}\} - 17\L\{ f(t)U(t-2) \}\\
						&= \frac{1}{s+5} - 17 e^{-2s}\L\{ f(t+2)\}\\
						&= \frac{1}{s+5} - 17 e^{-2s}\L\{ e^{-5t} + 2\}\\
						&= \frac{1}{s+5} - 17 e^{-2s}\left(\frac{1}{s+5} + \frac{2}{s} \right)\\
		\end{align*}
	
	\ben
	\item	What is wrong with his work?\\ 
			\vspace{1in}
			%\[ f(t+2) \ne e^{-5t} + 2 \]
	\item	Find the Laplace transform of the function $m(t).$\\
			\vspace{2.5in}
			%\[ M(s) = \frac{1}{s+5} - \frac{17 e^{-2s}e^{-10}}{s+5}  \]					
	\een

\noindent  {\it These problems are intended to help you review the Product Rule for differentiation.}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{RPPP}
\item    Write down the product rule.\\
	$\ds \frac{d}{dx}\Big( f(x)\cdot g(x) \Big) = \line(1,0){100}$

\item  Evaluate the following derivatives.
	\ben[label=(\alph*)]
	\item  $\ds \frac{d}{dx}\Big( e^x\cos x \Big)$
	\item  $\ds \frac{d}{dx}\Big( x^5\ln x \Big)$
	\item  $\ds \frac{d}{dx}\Big( x\arctan x \Big)$\\
	\een
	
\item  Given that $y$ is a function of $x$, use the product rule to rewrite the following:\\	
		$\ds \frac{d}{dx}\Big( \cos x \cdot y(x) \Big) = $\\
	
\item Given that $z$ is a function of $t$, use the product rule to rewrite the following:\\	
		$\ds \frac{d}{dt}\Big( e^{5t} \cdot z \Big) = $\\
		
\item  Now let's look at the product rule ``in the other direction"...  Or we can think about this
	as ``un-doing" the product rule.\\
	
	Rewrite each of the following as the derivative of a product:
	\ben[label=(\alph*)]
	\item  $\ds x^{-4} \frac{dy}{dx} - 4x^{-5}y = \frac{d}{dx}\Big( \hspace{2cm} \Big)$
	\item  $\ds e^{-3x} y' - 3e^{-3x}y = \frac{d}{dx}\Big( \hspace{2cm} \Big)$
	\item  $\ds \sec(\theta) \frac{dr}{d\theta} + \sec(\theta) \tan(\theta) r 
					= \frac{d}{d\theta}\Big( \hspace{2cm} \Big)$
	\een
	
\end{RPPP}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\showto{answers}{
\newpage
\noindent  Answers:
\begin{RPPP}
\item   product rule: $\ds \frac{d}{dx}\Big( f(x)\cdot g(x) \Big) = f(x)g'(x) + f'(x) g(x)$

\item  \ben[label=(\alph*)]
	\item  $\ds \frac{d}{dx}\Big( e^x\cos x \Big) = -e^x\sin x + e^x\cos x$\\
	\item  $\ds \frac{d}{dx}\Big( x^5\ln x \Big) = x^4 + 5x^4\ln x$\\
	\item  $\ds \frac{d}{dx}\Big( x\arctan x \Big) = \frac{x}{1+x^2} + \arctan x$\\
	\een
	
\item  $\ds \frac{d}{dx}\Big( \cos x \cdot y(x) \Big) = \cos x\cdot y'(x) - \sin x \cdot y(x)$\\
	
\item $\ds \frac{d}{dt}\Big( e^{5t} \cdot z \Big) = e^{5t}\cdot z' + 5e^{5t}\cdot z$\\
		
\item  \ben[label=(\alph*)]
	\item  $\ds x^{-4} \frac{dy}{dx} - 4x^{-5}y = \frac{d}{dx}\Big( x^{-4}\cdot y \Big)$
	\item  $\ds e^{-3x} y' - 3e^{-3x}y = \frac{d}{dx}\Big( e^{-3x}\cdot y \Big)$
	\item  $\ds \sec(\theta) \frac{dr}{d\theta} + \sec(\theta) \tan(\theta) r 
					= \frac{d}{d\theta}\Big( \sec(\theta)\cdot r \Big)$
	\een
	
\end{RPPP}
}

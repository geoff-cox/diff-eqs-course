\begin{minipage}[t]{.49\linewidth}

        \item[]	\deriv{\sin x}{\cos x}
        \item[]	\deriv{\csc x}{-\csc x\cot x}
        \item[]	\deriv{\tan x}{\sec^2 x}
        \item[]	\deriv{\sin^{-1}x}{\frac{1}{\sqrt{1-x^2}}}
        \item[]	\deriv{\csc^{-1}x}{\frac{-1}{|x|\sqrt{x^2-1}}}

\noindent\textbf{\large Anti-derivatives}

\vspace{0.5cm}

\begin{minipage}[t]{.5\linewidth}
    \begin{enumerate}
        \item[]	\myint{\cos x}{\sin x+C}
        \item[]	\myint{\tan x}{-\ln |\cos x|+C}
        \item[]	\myint{\csc x}{-\ln |\csc x+\cot x|+C}
        \item[]	\myint{\csc^2x}{-\cot x+C}
        \item[]	\myint{\csc x\cot x}{-\csc x+C}
        \item[]	\myint{\frac{1}{\sqrt{a^2-x^2}}}{\sin^{-1}\left(\frac xa\right)+C}
        \item[]	\myint{a^x}{\frac{1}{\ln a}\cdot a^x+C}
    \end{enumerate}
\end{minipage}
\begin{minipage}[t]{.5\linewidth}
    \begin{enumerate}\addtocounter{enumi}{11}
        \item[]	\myint{\sin x}{-\cos x+C}
        \item[]	\myint{\cot x}{\ln |\sin x|+C}
        \item[]	\myint{\sec x}{\ln |\sec x+\tan x|+C}
        \item[]	\myint{\sec^2 x}{\tan x+C}
        \item[]	\myint{\sec x\tan x}{\sec x+C}
        \item[]	\myint{\frac{1}{x^2+a^2}}{\frac1a\tan^{-1}\left(\frac xa\right)+C}
        \item[]	\myint{\frac{1}{x\sqrt{x^2-a^2}}}{\frac1a\sec^{-1}\left(\frac{|x|}{a}\right)+C}
    \end{enumerate}
\end{minipage}

    \noindent {\bf Table of Laplace Transforms\\}
    Let \(u(t)\) be the unit step function, \(a, b, c, \alpha, \beta\) be real numbers and \(n = 1, 2, 3, ...\)

    \begin{tabular}{|c|c|c|}
        \hline
        & $f(t)$                        
        & $F(s) = \L\{ f(t) \}$ \\
        \hline
        \hline
        L1.	& 1                             & $\ds \frac{1}{s},\ s>0$\\
        \hline
        L2.	& $e^{at}$                      & $\ds \frac{1}{s-a},\ s>a$\\
        \hline
        L3.	& $t^n$                         & $\ds \frac{n!}{s^{n+1}},\ s>0$\\
        \hline
        L4.	& $\sin (bt)$                   & $\ds \frac{b}{s^2 + b^2},\ s>0$\\
        \hline
        L5.	& $\cos(bt)$                    & $\ds \frac{s}{s^2 + b^2},\ s>0$\\
        \hline
        L6.	& $t^n e^{at}$                  & $\ds \frac{n!}{(s-a)^{n+1}},\ s>a$\\
        \hline
        L7.	& $e^{at} \sin(bt)$             & $\ds \frac{b}{(s-a)^2 + b^2},\ s>a$\\
        \hline
        L8.	& $e^{at} \cos(bt)$             & $\ds \frac{s-a}{(s-a)^2 + b^2},\ s>a$\\
        \hline
        L9.	& $\alpha f(t) + \beta g(t)$    & $\ds \alpha F(s) + \beta G(s)$\\
        \hline
        L10.	& $e^{at}f(t)$              & $F(s-a)$\\
        \hline
        L11.	& $f'(t)$                   & $sF(s) - f(0)$\\
        \hline
        L12.	& $f''(t)$                  & $s^2 F(s) - sf(0) - f'(0)$\\
        \hline
        L13.	& $f^{(n)}(t)$              & $s^n F(s) - s^{n-1}f(0) - s^{n-2}f'(0) - \cdots - f^{(n-1)}(0)$\\
        \hline
        L14.	& $t^n f(t)$                & $\ds (-1)^n\frac{d^n}{ds^n}\big( F(s) \big) = (-1)^nF^{(n)}(s)$\\
        \hline
        L15.	& $u_c(t) = u(t-c)$         & $\ds  \frac{e^{-cs}}{s}$\\
        \hline
        L16.	& $\ds f(t)\cdot u_c(t)$    & $\ds e^{-cs} \L\{ f(t+c) \}$\\
        \hline
        L17.	& $\ds f(t-c)\cdot u_c(t)$  & $\ds e^{-cs}F(s)$\\
        \hline
        \hline
    \end{tabular}

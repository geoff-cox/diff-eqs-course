\noindent {\bf Table of Laplace Transforms\\} \hspace{0.25cm}

\begin{tabular}{|c|c|c|}
\hline
	& $f(t)$ 								& $F(s) = \L\{ f(t) \}$ \\
\hline
\hline
L1.	& 1									& $\ds \frac{1}{s},  \hspace{0.25cm}s>0$\\
\hline
L2.	& $e^{at}$								& $\ds \frac{1}{s-a},  \hspace{0.25cm}s>a$\\
\hline
L3.	& $t^n, \hspace{0.25cm}n = 1, 2, \ldots$		& $\ds \frac{n!}{s^{n+1}},  \hspace{0.25cm}s>0$\\
\hline
L4.	& $\sin (kt)$							& $\ds \frac{k}{s^2 + k^2},  \hspace{0.25cm}s>0$\\
\hline
L5.	& $\cos(kt)$							& $\ds \frac{s}{s^2 + k^2},  \hspace{0.25cm}s>0$\\
\hline
L6.	& $\sinh (kt)$							& $\ds \frac{k}{s^2 - k^2},  \hspace{0.25cm}s>0$\\
\hline
L7.	& $\cosh(kt)$							& $\ds \frac{s}{s^2 - k^2},  \hspace{0.25cm}s>0$\\
\hline
L8.	& $t^n e^{at},  \hspace{0.25cm}n = 1, 2, \ldots$	& $\ds \frac{n!}{(s-a)^{n+1}},  \hspace{0.25cm}s>a$\\
\hline
L9.	& $e^{at} \sin(kt)$						& $\ds \frac{k}{(s-a)^2 + k^2},  \hspace{0.25cm}s>a$\\
\hline
L10.	& $e^{at} \cos(kt)$						& $\ds \frac{s-a}{(s-a)^2 + k^2},  \hspace{0.25cm}s>a$\\
\hline
L11.	& $e^{at}f(t)$							& $F(s-a)$\\
\hline
L12.	& $f'(t)$								& $sF(s) - f(0)$\\
\hline
L13.	& $f''(t)$								& $s^2 F(s) - sf(0) - f'(0)$\\
\hline
L14.	& $f^{(n)}(t)$							& $s^n F(s) - s^{n-1}f(0) - s^{n-2}f'(0) - \cdots - f^{(n-1)}(0)$\\
\hline
L15.	& $t^n f(t)$							& $\ds (-1)^n\frac{d^n}{ds^n}\big( F(s) \big) = (-1)^nF^{(n)}(s)$\\
\hline
%L9.	& $\alpha f(t) + \beta g(t)$					& $\ds \alpha F(s) + \beta G(s)$ ($\alpha, \beta$ are constants)\\
%\hline
%L15.	& $U(t-a)$								& $\ds  \frac{e^{-as}}{s}$\\
%\hline

%\hline
%L17.	& $\ds f(t-a)U(t-a)$						& $\ds e^{-as}F(s)$\\

\end{tabular}
